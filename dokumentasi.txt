    "password" : "123",
    "name" : "johnny"
}

Product Dengan Pagination :
http://localhost:8080/api/products/1?sort=name&order=desc
atau secara defult pagination by id
http://localhost:8080/api/products/1
method: GET
bearer token : JWT Token 

Product Dengan By Id :
http://localhost:8080/api/product/1
method: GET
bearer token : JWT Token 

Create Product :
http://localhost:8080/api/product
method: POST
bearer token : JWT Token 
form-data :
name : baju
description : baju bagus
price : 20000
image(file) : test.png

Update Product :
http://localhost:8080/api/product/1
method: PUT
bearer token : JWT Token 
form-data :
name : baju kemeja
description : baju nya lusuh
price : 20000
image(file) : OPTIONAL

Delete Product :
http://localhost:8080/api/product/1
method: DELETE
